-- =====================================================
-- Clodu Storage Database Schema
-- PostgreSQL 15+
-- =====================================================

-- Таблица подписок (справочник)
CREATE TABLE subscription (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    price INTEGER NOT NULL DEFAULT 0
);

-- Таблица пользователей
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    subscription_id INTEGER REFERENCES subscription(id) ON DELETE SET NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    is_deleted BOOLEAN NOT NULL DEFAULT FALSE
);

-- Таблица сессий пользователей
CREATE TABLE user_session (
    id SERIAL PRIMARY KEY,
    user_id INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    jwt_token TEXT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Таблица групп
CREATE TABLE groups (
    id SERIAL PRIMARY KEY,
    groupname VARCHAR(100) NOT NULL,
    owner_id INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP
);

-- Таблица участников групп
CREATE TABLE group_members (
    id SERIAL PRIMARY KEY,
    group_id INTEGER NOT NULL REFERENCES groups(id) ON DELETE CASCADE,
    user_id INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    rights INTEGER NOT NULL DEFAULT 1, -- 1=admin, 2=editor, 3=viewer
    added_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    left_at TIMESTAMP,
    UNIQUE(group_id, user_id)
);

-- Таблица логов групп
CREATE TABLE group_logs (
    id SERIAL PRIMARY KEY,
    group_id INTEGER NOT NULL REFERENCES groups(id) ON DELETE CASCADE,
    user_id INTEGER NOT NULL REFERENCES users(id) ON DELETE SET NULL,
    description TEXT NOT NULL,
    edited_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Таблица папок (поддерживает пользовательское и групповое владение)
CREATE TABLE folders (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    owner_id INTEGER NOT NULL,  -- может быть users.id или groups.id
    owner_type VARCHAR(10) NOT NULL CHECK (owner_type IN ('user', 'group')),
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP
);

-- Таблица файлов
CREATE TABLE files (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    user_id INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    type VARCHAR(100),
    size_bytes BIGINT NOT NULL DEFAULT 0,
    hash VARCHAR(64) NOT NULL,
    added_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP
);

-- Таблица связей папок и файлов (один файл может быть в нескольких папках)
CREATE TABLE folder_files (
    id SERIAL PRIMARY KEY,
    folder_id INTEGER NOT NULL REFERENCES folders(id) ON DELETE CASCADE,
    file_id INTEGER NOT NULL REFERENCES files(id) ON DELETE CASCADE,
    space_id INTEGER,  -- связь с пространством (опционально)
    UNIQUE(folder_id, file_id)
);

-- Таблица вложенности папок
CREATE TABLE folder_folders (
    id SERIAL PRIMARY KEY,
    parent_id INTEGER NOT NULL REFERENCES folders(id) ON DELETE CASCADE,
    child_id INTEGER NOT NULL REFERENCES folders(id) ON DELETE CASCADE,
    UNIQUE(parent_id, child_id)
);

-- Таблица тегов (ML-генерация)
CREATE TABLE tags (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL UNIQUE
);

-- Таблица связей файлов с тегами
CREATE TABLE file_tags (
    id SERIAL PRIMARY KEY,
    tag_id INTEGER NOT NULL REFERENCES tags(id) ON DELETE CASCADE,
    file_id INTEGER NOT NULL REFERENCES files(id) ON DELETE CASCADE,
    UNIQUE(tag_id, file_id)
);

-- Таблица пространств (проекции для ML)
CREATE TABLE spaces (
    id SERIAL PRIMARY KEY,
    owner_id INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(255) NOT NULL
);

-- Таблица связей пространств с тегами
CREATE TABLE space_tags (
    id SERIAL PRIMARY KEY,
    space_id INTEGER NOT NULL REFERENCES spaces(id) ON DELETE CASCADE,
    tag_id INTEGER NOT NULL REFERENCES tags(id) ON DELETE CASCADE,
    UNIQUE(space_id, tag_id)
);

-- =====================================================
-- Индексы для производительности
-- =====================================================

-- users
CREATE INDEX idx_users_username ON users(username);
CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_is_deleted ON users(is_deleted);

-- user_session
CREATE INDEX idx_user_session_user_id ON user_session(user_id);
CREATE INDEX idx_user_session_created_at ON user_session(created_at);

-- groups
CREATE INDEX idx_groups_owner_id ON groups(owner_id);
CREATE INDEX idx_groups_deleted_at ON groups(deleted_at);

-- group_members
CREATE INDEX idx_group_members_group_id ON group_members(group_id);
CREATE INDEX idx_group_members_user_id ON group_members(user_id);
CREATE INDEX idx_group_members_left_at ON group_members(left_at);

-- group_logs
CREATE INDEX idx_group_logs_group_id ON group_logs(group_id);
CREATE INDEX idx_group_logs_user_id ON group_logs(user_id);
CREATE INDEX idx_group_logs_edited_at ON group_logs(edited_at);

-- folders
CREATE INDEX idx_folders_owner ON folders(owner_id, owner_type);
CREATE INDEX idx_folders_deleted_at ON folders(deleted_at);
CREATE INDEX idx_folders_name ON folders(name);

-- files
CREATE INDEX idx_files_user_id ON files(user_id);
CREATE INDEX idx_files_deleted_at ON files(deleted_at);
CREATE INDEX idx_files_hash ON files(hash);
CREATE INDEX idx_files_added_at ON files(added_at);

-- folder_files
CREATE INDEX idx_folder_files_folder_id ON folder_files(folder_id);
CREATE INDEX idx_folder_files_file_id ON folder_files(file_id);
CREATE INDEX idx_folder_files_space_id ON folder_files(space_id);

-- folder_folders
CREATE INDEX idx_folder_folders_parent_id ON folder_folders(parent_id);
CREATE INDEX idx_folder_folders_child_id ON folder_folders(child_id);

-- tags
CREATE INDEX idx_tags_name ON tags(name);

-- file_tags
CREATE INDEX idx_file_tags_tag_id ON file_tags(tag_id);
CREATE INDEX idx_file_tags_file_id ON file_tags(file_id);

-- spaces
CREATE INDEX idx_spaces_owner_id ON spaces(owner_id);

-- space_tags
CREATE INDEX idx_space_tags_space_id ON space_tags(space_id);
CREATE INDEX idx_space_tags_tag_id ON space_tags(tag_id);
